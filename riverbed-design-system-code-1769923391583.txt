// Filter Toggle Pattern
const [showFilters, setShowFilters] = useState(false);
const [filters, setFilters] = useState({ name: '', status: '', type: '' });

// Count active filters
const activeFilterCount = Object.values(filters).filter(v => v !== '').length;

// Filter Toggle Button
<button
  onClick={() => setShowFilters(!showFilters)}
  className={`flex items-center gap-2 px-3 py-1.5 rounded transition-colors ${
    activeFilterCount > 0
      ? 'bg-purple-900 text-white'
      : 'bg-white text-gray-700 border border-gray-200 hover:bg-gray-50'
  }`}
>
  <Filter className="w-4 h-4" />
  <span className="text-sm">
    {activeFilterCount > 0 ? `Filters (${activeFilterCount})` : 'Filters'}
  </span>
</button>

// Grid Filter Inputs (above table)
{showFilters && (
  <div className="grid grid-cols-3 gap-4 mb-4">
    <Input
      placeholder="Filter by name..."
      value={filters.name}
      onChange={(e) => setFilters({ ...filters, name: e.target.value })}
    />
    <Input
      placeholder="Filter by status..."
      value={filters.status}
      onChange={(e) => setFilters({ ...filters, status: e.target.value })}
    />
    <Input
      placeholder="Filter by type..."
      value={filters.type}
      onChange={(e) => setFilters({ ...filters, type: e.target.value })}
    />
  </div>
)}


// ========================================
// FACETEDFILTER
// ========================================

// Faceted Filter Pattern
const [filters, setFilters] = useState<{
  type?: string[];
  division?: string[];
  channel?: string[];
}>({ type: undefined, division: undefined, channel: undefined });

// Toggle filter value
const toggleFilter = (category: 'type' | 'division' | 'channel', value: string) => {
  const current = filters[category] || [];
  const newValues = current.includes(value)
    ? current.filter(v => v !== value)
    : [...current, value];
  setFilters({
    ...filters,
    [category]: newValues.length > 0 ? newValues : undefined
  });
};

// Faceted Filter UI
<div className="space-y-4">
  {/* Type Filter */}
  <div>
    <h3 className="text-sm font-medium text-gray-700 mb-2">Type</h3>
    <div className="flex flex-wrap gap-2">
      <button
        onClick={() => setFilters({ ...filters, type: undefined })}
        className="px-3 py-1.5 text-sm rounded border border-gray-200 hover:bg-gray-50"
      >
        All
      </button>
      {['Notification', 'Survey', 'Confirmation'].map(type => (
        <button
          key={type}
          onClick={() => toggleFilter('type', type)}
          className={`px-3 py-1.5 text-sm rounded border transition-colors ${
            filters.type?.includes(type)
              ? 'bg-purple-50 border-purple-300 text-purple-900'
              : 'border-gray-200 hover:bg-gray-50'
          }`}
        >
          {type} <span className="text-gray-500">(5)</span>
        </button>
      ))}
    </div>
  </div>

  {/* Reset All */}
  {(filters.type?.length || filters.division?.length || filters.channel?.length) && (
    <button
      onClick={() => setFilters({ type: undefined, division: undefined, channel: undefined })}
      className="text-sm text-purple-700 hover:text-purple-800"
    >
      Reset All
    </button>
  )}
</div>
